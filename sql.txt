#预告影片信息表
CREATE TABLE `ec_movie_filmPlantInfo` (
  `Id` int(20) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `filmName` varchar(200) DEFAULT NULL COMMENT '影片名称',
  `filmName1` varchar(200) DEFAULT NULL COMMENT '影片名称(英文或汉语拼音)',
  `Cast` varchar(2000) DEFAULT NULL COMMENT '演员',
  `image` varchar(200) DEFAULT NULL COMMENT '海报',
  `background_image` varchar(200) DEFAULT NULL COMMENT '背景图片',
  `Version` varchar(200) DEFAULT NULL COMMENT '发行版本 分为 普通 普通立体  巨幕立体',
  `Duration` int(20) DEFAULT 0 COMMENT '影片时长，以分钟为单位',
  `PublishDate` varchar(200) DEFAULT NULL COMMENT '公映日期',
  `publishDateDesc` varchar(200) DEFAULT NULL COMMENT '公映日期  用于分组显示',
	`publishDateNum` varchar(200) DEFAULT NULL COMMENT '公映日期yyyymmdd 用于排序',
	`isAttention` int(2) DEFAULT 2 COMMENT '是否最受关注 1 最受关注 2 不受关注',
	`inaword` varchar(2000) DEFAULT NULL COMMENT '一句话描述',
	`filmReview` varchar(2000) DEFAULT NULL COMMENT '影评',
	`video` varchar(200) DEFAULT NULL COMMENT '预告片地址',
  `Director` varchar(200) DEFAULT NULL COMMENT '导演',
	`country` varchar(200) DEFAULT NULL COMMENT '国家',
	`filmType` varchar(200) DEFAULT NULL COMMENT '影片类型',
	`recommendIndex` int(2) DEFAULT 0 COMMENT '推荐指数  整数1-5',
	`starIndex` int(2) DEFAULT 0 COMMENT '明星阵容 整数 1-5',
	`seehearIndex` int(2) DEFAULT 0 COMMENT '视听效果 整数 1-5',
	`storyIndex` int(2) DEFAULT 0 COMMENT '剧情故事 整数 1-5',
	`mouthIndex` int(2) DEFAULT 0 COMMENT '观众口碑 整数 1-5',
	`watchingFocus` varchar(5000) DEFAULT NULL COMMENT '看点',
  `introduction` varchar(3000) DEFAULT NULL COMMENT '内容简介',
  PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=100 DEFAULT CHARSET=utf8 COMMENT='预告影片信息表';
